###    HELP DESK
We have created an interface for our helpdesk in which users will be able to post their queries.
We have created Query form in which user will enter his/her personal details along with the query.
We enable user to know more about the college and provided them with college helpline number along with email support.

### FEATURES
1)Created  help desk which allows the user to post his/her queries regarding college by clicking on CONTACT US option.

2)The help desk should provide ldap login to the user along with guest login(where the user have to register) the user have to provide the tag for his/her query along with the concern department to whom the query belongs.

3)The help desk should have an admin panel where the admin can see all the queries rased by the users (at backend service we can see the quiries at admin panel).

4)Authorized person of concerned department forward the query to concerned person and HoD can see all the queries raised to their department

5)HoD can also forward the query to the concerned teacher.

6)All the queries and their solutions will not be directly posted to fornt page of the Helpdesk only the queries approved by HOD or admin will be posted as FAQ on help desk portal at backened.

7)Confirmation mail is sent to the users of their enquiry.

      $(function () {
          var KnowledgebaseFolderModel = Backbone.Model.extend({
              validation: {
                'title': [{
                  required: true,
                  msg: "Please specify a title for the folder."
                }, {
                  pattern: '^((?![$%<]).)*$',
                  msg: "The folder title must include valid characters only."
                }]
              }
          });

          var KnowledgebaseFolderCategoryModel = Backbone.Model.extend({
              idAttribute: "id"
          });

          var KnowledgebaseFolderArticleModel = Backbone.Model.extend({
              idAttribute: "id"
          });

          var KnowledgebaseFolderArticleCollection = AppCollection.extend({
              model: KnowledgebaseFolderArticleModel,
              url: "/en/member/knowledgebase/folders/702/xhr/articles",
              initialize: function(params) {
                  this.parent = params.parent;
              },
              parseRecords: function (resp, options) {
                  return resp.articles;
              },
              load: function(currentPage, sortKey, order) {
                  this.state.order = order;
                  this.state.sortKey = sortKey;
                  this.state.currentPage = currentPage;

                  var node = $('.uv-tab-active');

                  if (node.data('section') != 'articles') {
                      node.removeClass('uv-tab-active');
                      $('li[data-section="articles"]').addClass('uv-tab-active');
                  }

                  if ($('.uv-kb-folder-article-listing').length == 0) {
                      $('.uv-kb-section').html(this.parent.folder_section_loading_template());
                  }

                  this.fetch({
                      reset: true,
                      success: (model, response) => {
                          if ($('.uv-kb-folder-article-listing').length == 0) {
                              $('.uv-kb-section').html(this.parent.folder_tagged_articles_template({
                                  pagination: response.pagination
                              }));
                          }

                          app.pager.paginationData = response.pagination;
                          var url = app.pager.paginationData.url;

                          if (response.length == 0 && app.pager.paginationData.current != "0")
                              router.navigate(url.replace('replacePage', app.pager.paginationData.last), { trigger: true });
                          else {
                              app.pager.render();
                          }

                          $('.uv-kb-section .collection').html(this.parent.folder_tagged_articles_collection_template({
                              articles: response.articles
                          }));
                      },
                      error: function (model, xhr, options) {
                          if (url = xhr.getResponseHeader('Location'))
                              window.location = url;
                      }
                  });
              }
          });

          var KnowledgebaseFolderCategoryCollection = AppCollection.extend({
              model: KnowledgebaseFolderCategoryModel,
              url: "/en/member/knowledgebase/folders/702/xhr/categories",
              initialize: function(params) {
                  this.parent = params.parent;
              },
              parseRecords: function (resp, options) {
                  return resp.folders;
              },
              load: function(currentPage, sortKey, order) {
                  this.state.order = order;
                  this.state.sortKey = sortKey;
                  this.state.currentPage = currentPage;

                  var node = $('.uv-tab-active');

                  if (node.data('section') != 'categories') {
                      node.removeClass('uv-tab-active');

                  }

                  if ($('.uv-kb-folder-category-listing').length == 0) {
                      $('.uv-kb-section').html(this.parent.folder_section_loading_template());

                      success: (model, response) => {

                              $('.uv-kb-section').html(this.parent.folder_tagged_categories_template({
                                  pagination: response.pagination

                          if (response.length == 0 && app.pager.paginationData.current != "0")
                             router.navigate(url.replace('replacePage', app.pager.paginationData.last), { trigger: true });
                          else {
                              app.pager.render();
                          }

                          $('.uv-kb-section .collection').html(this.parent.folder_tagged_categories_collection_template({
                              categories: response.categories
                          }));
                      },
                      error: (model, xhr, options) => {
                          if (url = xhr.getResponseHeader('Location')) {
                              window.location = url;
                          }
                      }

              }
          });

          var Knowledgebase = Backbone.View.extend({
              folder_details_template: _.template($("#knowledgebase_folder_details_template").html()),
              folder_section_loading_template: _.template($("#knowledgebase_folder_section_loading_template").html()),
              folder_tagged_categories_template: _.template($("#knowledgebase_folder_tagged_categories_template").html()),
              folder_tagged_articles_template: _.template($("#knowledgebase_folder_tagged_articles_template").html()),
              folder_tagged_categories_collection_template: _.template($("#knowledgebase_folder_tagged_categories_collection_template").html()),
              folder_tagged_articles_collection_template: _.template($("#knowledgebase_folder_tagged_articles_collection_template").html()),
              loader_template: _.template($("#loader-tmp").html()),

              events : {
                  'click .uv-kb-tabs li': 'navigateTab',
                  'blur input, textarea': 'formChanged',
                  'submit #article-form': 'updateCategoryDetails',
                  'click #folder-status-pta .uv-dropdown-list li': 'updatePublishingStatus',
              },

              initialize : function(params) {
                  this.articles = new KnowledgebaseFolderArticleCollection({
                      parent: this
                  });

                  this.categories = new KnowledgebaseFolderCategoryCollection({
                      parent: this
                  });

                  Backbone.Validation.bind(this);
                  var jsonContext = JSON.parse('[]');

                    for (var field in jsonContext) {
                      Backbone.Validation.callbacks.invalid(this, field, jsonContext[field], 'input');
                  }
              },
              formChanged: function(e) {
                  this.model.set(Backbone.$(e.currentTarget).attr('name'), Backbone.$(e.currentTarget).val())
                  this.model.isValid([Backbone.$(e.currentTarget).attr('name')])
              },
              updatePublishingStatus: function (e) {
                  var target = $(e.target);
                  var isPublished = target.data('index');

                  var url = "/en/member/knowledgebase/folders/702/xhr/status";

                  $.post(url, {
                      isPublished: isPublished
                  }, function(response) {
                      $('#folder-status-pta .uv-dropdown-list li').removeAttr('selected');
                      target.attr('selected', 'selected');

                      if (response.content.isPublished == true) {
                          $('#folder-status-pta .uv-aside-select-value').data('id', 1);
                          $('#folder-status-pta .uv-aside-select-value').html('Published');
                      } else {
                          $('#folder-status-pta .uv-aside-select-value').data('id', 0);
                          $('#folder-status-pta .uv-aside-select-value').html('Draft');
                      }
                  });
              },
              updateCategoryDetails: function(e) {
                  this.model.set($('#article-form').serializeObject());

                  if (false == this.model.isValid(true)) {
                      e.preventDefault();
                  }
              },
              navigateTab: function(e) {
                  var target = $(e.target);
                  var current = $('.uv-tab-active');

                  if (target[0].tagName == 'I') {
                      target = $(target[0]).parent();
                  }

                  var targetSection = target.data('section');
                  var currentSection = current.data('section');

                  if (currentSection != targetSection) {
                      var path = (targetSection == 'details') ? '#' : '#' + target.data('section');
                      router.navigate(path, { trigger: true });
                  }
              },
              renderFolderDetails: function() {
                  var node = $('.uv-tab-active');

                  if (node.data('section') != 'details') {
                      node.removeClass('uv-tab-active');
                      $('li[data-section="details"]').addClass('uv-tab-active');
                  }

                  $('.uv-kb-section').html(this.folder_details_template());
              }
          });

              },

                  this.knowledgebase = params.knowledgebase;
              },
              knowledgebaseFolderDetails: function() {
